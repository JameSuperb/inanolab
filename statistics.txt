        // ============= Sidebar Toggle =============
        function toggleSidebar() {
            const sidebar = document.getElementById('sidebar');
            const button = event.target.closest('.sidebar-toggle');
            const icon = button.querySelector('.sidebar-toggle-icon');
            sidebar.classList.toggle('collapsed');
            icon.textContent = sidebar.classList.contains('collapsed') ? '»' : '«';
        }

        // ============= Tab Management =============
        function switchTab(tabName) {
            document.querySelectorAll('.tab-content').forEach(tab => tab.classList.remove('active'));
            document.querySelectorAll('.tab-btn').forEach(btn => btn.classList.remove('active'));
            
            document.getElementById(tabName).classList.add('active');
            event.target.classList.add('active');
        }

        // ============= Column Header Management =============
        function renderColumnHeaderInputs() {
            const schema = testSchemas[currentTest];
            const container = document.getElementById('column-headers-input');
            let html = '';
            
            schema.columns.forEach(col => {
                html += `
                    <div class="column-input-group">
                        <label>${col}</label>
                        <input type="text" id="header-${col}" value="${col}" placeholder="Enter header name">
                    </div>
                `;
            });
            
            container.innerHTML = html;
        }

        function applyColumnHeaders() {
            const schema = testSchemas[currentTest];
            const headers = {};
            
            schema.columns.forEach(col => {
                const input = document.getElementById(`header-${col}`);
                headers[col] = input?.value || col;
            });
            
            // Store custom headers
            window.customHeaders = headers;
            
            // Render data entry table with new headers
            renderDataEntryTable();
        }

        function renderDataEntryTable() {
            const schema = testSchemas[currentTest];
            const container = document.getElementById('data-table-container');
            
            if (!window.customHeaders) {
                renderColumnHeaderInputs();
                applyColumnHeaders();
                return;
            }
            
            const headers = window.customHeaders;
            const columns = schema.columns;
            
            let html = '<table class="data-entry-table"><thead><tr>';
            columns.forEach(col => {
                html += `<th>${headers[col] || col}</th>`;
            });
            html += '<th style="width: 40px;"></th></tr></thead><tbody>';
            
            if (currentData.length === 0) {
                // Show one empty row
                html += '<tr>';
                columns.forEach((col, idx) => {
                    html += `<td><input type="text" data-column="${col}" data-row="0" onchange="updateDataCell(0, '${col}', event)"></td>`;
                });
                html += '<td><button class="btn" style="padding: 0.25rem 0.5rem; font-size: 0.75rem;" onclick="deleteDataRow(0)">×</button></td></tr>';
            } else {
                currentData.forEach((row, rowIdx) => {
                    html += '<tr>';
                    columns.forEach(col => {
                        const value = row[col] || '';
                        html += `<td><input type="text" data-column="${col}" data-row="${rowIdx}" value="${value}" onchange="updateDataCell(${rowIdx}, '${col}', event)"></td>`;
                    });
                    html += `<td><button class="btn" style="padding: 0.25rem 0.5rem; font-size: 0.75rem;" onclick="deleteDataRow(${rowIdx})">×</button></td></tr>`;
                });
            }
            
            html += '</tbody></table>';
            container.innerHTML = html;
        }

        function updateDataCell(rowIdx, column, event) {
            const value = event.target.value;
            const numValue = isNaN(value) ? value : parseFloat(value);
            
            if (!currentData[rowIdx]) {
                currentData[rowIdx] = {};
            }
            currentData[rowIdx][column] = numValue;
        }

        function addDataRow() {
            const newRow = {};
            const schema = testSchemas[currentTest];
            schema.columns.forEach(col => {
                newRow[col] = '';
            });
            currentData.push(newRow);
            renderDataEntryTable();
        }

        function deleteDataRow(rowIdx) {
            currentData.splice(rowIdx, 1);
            renderDataEntryTable();
        }

        function clearAllData() {
            if (currentData.length === 0) return;
            currentData = [];
            renderDataEntryTable();
            document.getElementById('results-panel').style.display = 'none';
            document.getElementById('no-results-message').style.display = 'block';
        }

        function determineRecommendation() {
            const factors = document.querySelector('input[name="factors"]:checked')?.value;
            let recommended = null;
            let why = '';
            let alternative = null;

            if (factors === 'one') {
                const observations = document.querySelector('input[name="observations"]:checked')?.value;
                const condCount = document.querySelector('input[name="condition-count"]:checked')?.value;
                const normality = document.querySelector('input[name="normality"]:checked')?.value;

                if (observations && condCount && normality) {
                    if (observations === 'independent') {
                        if (condCount === 'two') {
                            recommended = normality === 'yes' ? 'ttest-ind' : 'mann-whitney';
                        } else if (condCount === 'three-plus') {
                            recommended = normality === 'yes' ? 'anova-one' : 'kruskal-wallis';
                        }
                    }
                    else if (observations === 'paired') {
                        if (condCount === 'two') {
                            recommended = normality === 'yes' ? 'ttest-paired' : 'wilcoxon';
                        } else if (condCount === 'three-plus') {
                            recommended = normality === 'yes' ? 'anova-rm' : 'friedman';
                        }
                    }
                    else if (observations === 'repeated') {
                        if (condCount === 'two') {
                            recommended = normality === 'yes' ? 'ttest-paired' : 'wilcoxon';
                        } else if (condCount === 'three-plus') {
                            recommended = normality === 'yes' ? 'anova-rm' : 'friedman';
                        }
                    }
                }
            } else if (factors === 'two') {
                const design = document.querySelector('input[name="design"]:checked')?.value;
                const normality2 = document.querySelector('input[name="normality-2factor"]:checked')?.value;

                if (design && normality2) {
                    if (design === 'independent-only') {
                        if (normality2 === 'yes') {
                            recommended = 'anova-two';
                        } else {
                            document.getElementById('recommendation-test-name').textContent = 'Not Supported';
                            document.getElementById('recommendation-why').textContent = '⚠️ Non-parametric two-way ANOVA is not supported in this version.';
                            document.getElementById('recommendation-alternative').textContent = '💡 Try: Validate your normality assumption or use parametric test.';
                            document.getElementById('recommendation-display').style.display = 'block';
                            document.getElementById('use-recommended-btn').style.display = 'none';
                            return;
                        }
                    } else if (design === 'includes-repeated') {
                        document.getElementById('recommendation-test-name').textContent = 'Not Supported';
                        document.getElementById('recommendation-why').textContent = '⚠️ Mixed designs with repeated measurements are not supported yet.';
                        document.getElementById('recommendation-alternative').textContent = '💡 Tip: Consult a statistician for mixed model analysis.';
                        document.getElementById('recommendation-display').style.display = 'block';
                        document.getElementById('use-recommended-btn').style.display = 'none';
                        return;
                    }
                }
            }

            if (recommended) {
                recommendedTest = recommended;
                const metadata = testMetadata[recommended];
                document.getElementById('recommendation-test-name').textContent = metadata.name;
                document.getElementById('recommendation-why').textContent = '📋 ' + metadata.why;
                
                if (metadata.alternative) {
                    document.getElementById('recommendation-alternative').textContent = '💡 Alternative: ' + metadata.alternative;
                } else {
                    document.getElementById('recommendation-alternative').textContent = '';
                }

                document.getElementById('recommendation-display').style.display = 'block';
                document.getElementById('use-recommended-btn').style.display = 'inline-block';
            } else {
                document.getElementById('recommendation-display').style.display = 'none';
                document.getElementById('use-recommended-btn').style.display = 'none';
            }
        }

        function handleCSVUpload(event) {
            const file = event.target.files[0];
            if (!file) return;

            const reader = new FileReader();
            reader.onload = (e) => {
                try {
                    const csv = e.target.result;
                    const lines = csv.trim().split('\n');
                    const headers = lines[0].split(',').map(h => h.trim());
                    const data = [];

                    for (let i = 1; i < lines.length; i++) {
                        if (lines[i].trim() === '') continue;
                        const values = lines[i].split(',').map(v => v.trim());
                        const row = {};
                        headers.forEach((header, idx) => {
                            row[header] = isNaN(values[idx]) ? values[idx] : parseFloat(values[idx]);
                        });
                        data.push(row);
                    }

                    currentData = data;
                    renderDataEntryTable();
                    runTest();
                } catch (error) {
                    alert('Error parsing CSV: ' + error.message);
                }
            };
            reader.readAsText(file);
        }

        // ============= Statistical Tests =============
        function validateData(requiredColumns) {
            if (currentData.length === 0) return false;
            const firstRow = currentData[0];
            return requiredColumns.every(col => col in firstRow);
        }

        function runTest() {
            const schema = testSchemas[currentTest];

            if (!validateData(schema.columns)) {
                showResults({
                    status: 'error',
                    message: 'Invalid data schema. ' + schema.name + ' requires: ' + schema.columns.join(', ')
                });
                return;
            }

            let results = null;
            try {
                switch (currentTest) {
                    case 'ttest-ind':
                        results = computeTtestIndependent();
                        break;
                    case 'ttest-paired':
                        results = computeTtestPaired();
                        break;
                    case 'anova-one':
                        results = computeAnovaOne();
                        break;
                    case 'anova-two':
                        results = computeAnovaTwo();
                        break;
                    case 'anova-rm':
                        results = computeAnovaRM();
                        break;
                    case 'mann-whitney':
                        results = computeMannWhitney();
                        break;
                    case 'wilcoxon':
                        results = computeWilcoxon();
                        break;
                    case 'kruskal-wallis':
                        results = computeKruskalWallis();
                        break;
                    case 'friedman':
                        results = computeFriedman();
                        break;
                }
                currentResults = results;
                showResults(results);
            } catch (error) {
                showResults({
                    status: 'error',
                    message: 'Computation error: ' + error.message
                });
            }
        }

        function computeTtestIndependent() {
            const groups = groupBy(currentData, 'group');
            const groupA = groups[Object.keys(groups)[0]].map(d => d.value);
            const groupB = groups[Object.keys(groups)[1]].map(d => d.value);

            const meanA = jStat.mean(groupA);
            const meanB = jStat.mean(groupB);
            const n1 = groupA.length;
            const n2 = groupB.length;

            const varA = jStat.variance(groupA, true);
            const varB = jStat.variance(groupB, true);

            const pooledVar = ((n1 - 1) * varA + (n2 - 1) * varB) / (n1 + n2 - 2);
            const se = Math.sqrt(pooledVar * (1 / n1 + 1 / n2));
            const t = (meanA - meanB) / se;
            const df = n1 + n2 - 2;
            const pValue = 2 * (1 - jStat.studentt.cdf(Math.abs(t), df));

            const cohensD = (meanA - meanB) / Math.sqrt(pooledVar);

            return {
                status: 'success',
                testName: 'Independent t-test',
                statistic: t.toFixed(4),
                statisticLabel: 't',
                df: df,
                pValue: pValue.toFixed(4),
                effectSize: cohensD.toFixed(4),
                effectSizeLabel: "Cohen's d",
                significant: pValue < 0.05
            };
        }

        function computeTtestPaired() {
            const subjects = groupBy(currentData, 'subject');
            const diffs = [];

            Object.values(subjects).forEach(measures => {
                const pre = measures.find(m => 'pre' in m).pre;
                const post = measures.find(m => 'post' in m).post;
                diffs.push(post - pre);
            });

            const meanDiff = jStat.mean(diffs);
            const seDiff = jStat.stdev(diffs, true) / Math.sqrt(diffs.length);
            const t = meanDiff / seDiff;
            const df = diffs.length - 1;
            const pValue = 2 * (1 - jStat.studentt.cdf(Math.abs(t), df));

            const cohensD = meanDiff / jStat.stdev(diffs, true);

            return {
                status: 'success',
                testName: 'Paired t-test',
                statistic: t.toFixed(4),
                statisticLabel: 't',
                df: df,
                pValue: pValue.toFixed(4),
                effectSize: cohensD.toFixed(4),
                effectSizeLabel: "Cohen's d",
                significant: pValue < 0.05
            };
        }

        function computeAnovaOne() {
            const groups = groupBy(currentData, 'group');
            const groupArrays = Object.values(groups).map(g => g.map(d => d.value));

            const allValues = groupArrays.flat();
            const grandMean = jStat.mean(allValues);

            const k = groupArrays.length;
            const N = allValues.length;

            let ss_between = 0;
            groupArrays.forEach(group => {
                const groupMean = jStat.mean(group);
                ss_between += group.length * Math.pow(groupMean - grandMean, 2);
            });

            let ss_within = 0;
            groupArrays.forEach(group => {
                const groupMean = jStat.mean(group);
                group.forEach(val => {
                    ss_within += Math.pow(val - groupMean, 2);
                });
            });

            const df_between = k - 1;
            const df_within = N - k;
            const ms_between = ss_between / df_between;
            const ms_within = ss_within / df_within;
            const F = ms_between / ms_within;
            const pValue = 1 - jStat.centralF.cdf(F, df_between, df_within);

            const eta2 = ss_between / (ss_between + ss_within);

            return {
                status: 'success',
                testName: 'One-way ANOVA',
                statistic: F.toFixed(4),
                statisticLabel: 'F',
                df: `${df_between}, ${df_within}`,
                pValue: pValue.toFixed(4),
                effectSize: eta2.toFixed(4),
                effectSizeLabel: 'η²',
                significant: pValue < 0.05
            };
        }

        function computeAnovaTwo() {
            const groups = groupBy(currentData, d => d.factorA + '-' + d.factorB);
            const allValues = currentData.map(d => d.value);
            const grandMean = jStat.mean(allValues);
            const N = allValues.length;

            let ss_total = 0;
            allValues.forEach(val => {
                ss_total += Math.pow(val - grandMean, 2);
            });

            let ss_between = 0;
            Object.values(groups).forEach(group => {
                const groupMean = jStat.mean(group.map(d => d.value));
                group.forEach(d => {
                    ss_between += Math.pow(groupMean - grandMean, 2);
                });
            });

            const ss_within = ss_total - ss_between;
            const df_between = Object.keys(groups).length - 1;
            const df_within = N - Object.keys(groups).length;

            const ms_between = ss_between / df_between;
            const ms_within = ss_within / df_within;
            const F = ms_between / ms_within;
            const pValue = 1 - jStat.centralF.cdf(F, df_between, df_within);

            const eta2 = ss_between / ss_total;

            return {
                status: 'success',
                testName: 'Two-way ANOVA',
                statistic: F.toFixed(4),
                statisticLabel: 'F',
                df: `${df_between}, ${df_within}`,
                pValue: pValue.toFixed(4),
                effectSize: eta2.toFixed(4),
                effectSizeLabel: 'η²',
                significant: pValue < 0.05
            };
        }

        function computeAnovaRM() {
            const subjects = groupBy(currentData, 'subject');
            const conditions = new Set(currentData.map(d => d.condition));

            const subjectCount = Object.keys(subjects).length;
            const conditionCount = conditions.size;
            const N = currentData.length;

            const allValues = currentData.map(d => d.value);
            const grandMean = jStat.mean(allValues);

            let ss_total = 0;
            allValues.forEach(val => {
                ss_total += Math.pow(val - grandMean, 2);
            });

            let ss_between_subjects = 0;
            Object.values(subjects).forEach(subject => {
                const subjectMean = jStat.mean(subject.map(d => d.value));
                subject.forEach(d => {
                    ss_between_subjects += Math.pow(subjectMean - grandMean, 2);
                });
            });

            let ss_conditions = 0;
            Array.from(conditions).forEach(cond => {
                const condValues = currentData.filter(d => d.condition === cond).map(d => d.value);
                const condMean = jStat.mean(condValues);
                condValues.forEach(val => {
                    ss_conditions += Math.pow(condMean - grandMean, 2);
                });
            });

            const ss_error = ss_total - ss_between_subjects - ss_conditions;
            const df_conditions = conditionCount - 1;
            const df_error = (subjectCount - 1) * (conditionCount - 1);

            const ms_conditions = ss_conditions / df_conditions;
            const ms_error = ss_error / df_error;
            const F = ms_conditions / ms_error;
            const pValue = 1 - jStat.centralF.cdf(F, df_conditions, df_error);

            const partialEta2 = ss_conditions / (ss_conditions + ss_error);

            return {
                status: 'success',
                testName: 'Repeated Measures ANOVA',
                statistic: F.toFixed(4),
                statisticLabel: 'F',
                df: `${df_conditions}, ${df_error}`,
                pValue: pValue.toFixed(4),
                effectSize: partialEta2.toFixed(4),
                effectSizeLabel: 'Partial η²',
                significant: pValue < 0.05
            };
        }

        function computeMannWhitney() {
            const groups = groupBy(currentData, 'group');
            const groupA = groups[Object.keys(groups)[0]].map(d => d.value).sort((a, b) => a - b);
            const groupB = groups[Object.keys(groups)[1]].map(d => d.value).sort((a, b) => a - b);

            const allValues = [...groupA, ...groupB].sort((a, b) => a - b);
            const ranks = {};
            let rank = 1;
            allValues.forEach((val, idx) => {
                if (idx === 0 || val !== allValues[idx - 1]) {
                    ranks[val] = rank;
                }
                rank++;
            });

            let R1 = 0;
            groupA.forEach(val => {
                R1 += ranks[val];
            });

            const n1 = groupA.length;
            const n2 = groupB.length;
            const U = n1 * n2 + (n1 * (n1 + 1)) / 2 - R1;

            const meanU = (n1 * n2) / 2;
            const stdU = Math.sqrt((n1 * n2 * (n1 + n2 + 1)) / 12);
            const z = Math.abs((U - meanU) / stdU);
            const pValue = 2 * (1 - jStat.normal.cdf(z, 0, 1));

            const r = z / Math.sqrt(n1 + n2);

            return {
                status: 'success',
                testName: 'Mann–Whitney U',
                statistic: U.toFixed(2),
                statisticLabel: 'U',
                pValue: pValue.toFixed(4),
                effectSize: r.toFixed(4),
                effectSizeLabel: 'r',
                significant: pValue < 0.05
            };
        }

        function computeWilcoxon() {
            const subjects = groupBy(currentData, 'subject');
            const diffs = [];

            Object.values(subjects).forEach(measures => {
                const pre = measures.find(m => 'pre' in m).pre;
                const post = measures.find(m => 'post' in m).post;
                const diff = post - pre;
                if (diff !== 0) diffs.push(diff);
            });

            const absDiffs = diffs.map(d => Math.abs(d)).sort((a, b) => a - b);
            const ranks = {};
            let rank = 1;
            absDiffs.forEach((val, idx) => {
                if (idx === 0 || val !== absDiffs[idx - 1]) {
                    ranks[val] = rank;
                }
                rank++;
            });

            let T = 0;
            diffs.forEach(diff => {
                if (diff > 0) {
                    T += ranks[Math.abs(diff)];
                }
            });

            const n = diffs.length;
            const meanT = (n * (n + 1)) / 4;
            const stdT = Math.sqrt((n * (n + 1) * (2 * n + 1)) / 24);
            const z = Math.abs((T - meanT) / stdT);
            const pValue = 2 * (1 - jStat.normal.cdf(z, 0, 1));

            const r = z / Math.sqrt(2 * n);

            return {
                status: 'success',
                testName: 'Wilcoxon Signed-Rank',
                statistic: T.toFixed(2),
                statisticLabel: 'T',
                pValue: pValue.toFixed(4),
                effectSize: r.toFixed(4),
                effectSizeLabel: 'r',
                significant: pValue < 0.05
            };
        }

        function computeKruskalWallis() {
            const groups = groupBy(currentData, 'group');
            const groupArrays = Object.values(groups).map(g => g.map(d => d.value));

            const allValues = groupArrays.flat().sort((a, b) => a - b);
            const ranks = {};
            let rank = 1;
            allValues.forEach((val, idx) => {
                if (idx === 0 || val !== allValues[idx - 1]) {
                    ranks[val] = rank;
                }
                rank++;
            });

            const N = allValues.length;
            let H = 0;

            groupArrays.forEach(group => {
                let R = 0;
                group.forEach(val => {
                    R += ranks[val];
                });
                H += Math.pow(R, 2) / group.length;
            });

            H = (12 / (N * (N + 1))) * H - 3 * (N + 1);

            const k = groupArrays.length;
            const df = k - 1;
            const pValue = 1 - jStat.chisquare.cdf(H, df);

            return {
                status: 'success',
                testName: 'Kruskal–Wallis',
                statistic: H.toFixed(4),
                statisticLabel: 'H',
                df: df,
                pValue: pValue.toFixed(4),
                effectSize: 'N/A',
                effectSizeLabel: '',
                significant: pValue < 0.05
            };
        }

        function computeFriedman() {
            const subjects = groupBy(currentData, 'subject');
            const conditions = new Set(currentData.map(d => d.condition));

            const conditionArray = Array.from(conditions).sort();
            const n = Object.keys(subjects).length;
            const k = conditionArray.length;

            let R = {};
            conditionArray.forEach(cond => {
                R[cond] = 0;
            });

            Object.values(subjects).forEach(subject => {
                const subjectValues = {};
                subject.forEach(d => {
                    subjectValues[d.condition] = d.value;
                });

                const values = conditionArray.map(c => subjectValues[c]).sort((a, b) => a - b);
                const ranks = {};
                values.forEach((val, idx) => {
                    ranks[val] = idx + 1;
                });

                conditionArray.forEach(cond => {
                    R[cond] += ranks[subjectValues[cond]];
                });
            });

            let sumR2 = 0;
            Object.values(R).forEach(r => {
                sumR2 += Math.pow(r, 2);
            });

            const X2 = (12 / (n * k * (k + 1))) * sumR2 - 3 * n * (k + 1);

            const df = k - 1;
            const pValue = 1 - jStat.chisquare.cdf(X2, df);

            return {
                status: 'success',
                testName: 'Friedman Test',
                statistic: X2.toFixed(4),
                statisticLabel: 'χ²',
                df: df,
                pValue: pValue.toFixed(4),
                effectSize: 'N/A',
                effectSizeLabel: '',
                significant: pValue < 0.05
            };
        }

        function generateSummaryTable() {
            const groups = new Set(currentData.map(d => d.group || d.factorA || d.subject || 'All'));
            let html = '<table class="publication-table"><thead><tr><th>Group</th><th>n</th><th>Mean</th><th>SD</th><th>Min</th><th>Max</th></tr></thead><tbody>';

            if (groups.has('All') || groups.size === 1) {
                const values = currentData.map(d => d.value).filter(v => v !== undefined);
                const n = values.length;
                const mean = jStat.mean(values);
                const sd = jStat.stdev(values, true);
                const min = Math.min(...values);
                const max = Math.max(...values);
                html += `<tr><td><strong>All Data</strong></td><td>${n}</td><td>${mean.toFixed(2)}</td><td>${sd.toFixed(2)}</td><td>${min.toFixed(2)}</td><td>${max.toFixed(2)}</td></tr>`;
            } else {
                Array.from(groups).forEach(group => {
                    const groupData = currentData.filter(d => (d.group || d.factorA || d.subject) === group);
                    const values = groupData.map(d => d.value).filter(v => v !== undefined);
                    if (values.length > 0) {
                        const n = values.length;
                        const mean = jStat.mean(values);
                        const sd = jStat.stdev(values, true);
                        const min = Math.min(...values);
                        const max = Math.max(...values);
                        html += `<tr><td><strong>${group}</strong></td><td>${n}</td><td>${mean.toFixed(2)}</td><td>${sd.toFixed(2)}</td><td>${min.toFixed(2)}</td><td>${max.toFixed(2)}</td></tr>`;
                    }
                });
            }

            html += '</tbody></table>';
            return html;
        }

        function generateChart(results) {
            const container = document.getElementById('chart-container');
            if (!container) return;

            const groups = groupBy(currentData, 'group');
            const groupNames = Object.keys(groups).sort();

            if (groupNames.length > 0 && groupNames[0] !== 'undefined') {
                // Mean with 95% CI and Median plot
                const groupStats = groupNames.map((name) => {
                    const data = groups[name];
                    const values = data.map(d => d.value).sort((a, b) => a - b);
                    const n = values.length;
                    const mean = jStat.mean(values);
                    const median = values[Math.floor(n / 2)];
                    const sd = jStat.stdev(values, true);
                    const se = sd / Math.sqrt(n);
                    const tCritical = 1.96; // 95% CI for large n
                    const ci95 = tCritical * se;

                    return {
                        name,
                        mean,
                        median,
                        ciLower: mean - ci95,
                        ciUpper: mean + ci95,
                        sd,
                        n,
                        se
                    };
                });

                // Destroy existing chart if it exists
                if (window.chartInstance) {
                    window.chartInstance.destroy();
                }

                // Clear container and create canvas
                container.innerHTML = '<canvas></canvas>';
                const canvas = container.querySelector('canvas');
                const ctx = canvas.getContext('2d');
                
                // Plugin to draw CI error bars with proper styling
                const ciErrorBarsPlugin = {
                    id: 'ciErrorBars',
                    afterDatasetsDraw(chart) {
                        const ctx = chart.ctx;
                        const xScale = chart.scales.x;
                        const yScale = chart.scales.y;

                        groupStats.forEach((group, idx) => {
                            const x = xScale.getPixelForValue(idx);
                            const yMean = yScale.getPixelForValue(group.mean);
                            const yUpper = yScale.getPixelForValue(group.ciUpper);
                            const yLower = yScale.getPixelForValue(group.ciLower);

                            // Draw thin vertical line for CI in light blue
                            ctx.strokeStyle = '#7CB9E8';
                            ctx.lineWidth = 1.5;
                            ctx.beginPath();
                            ctx.moveTo(x, yUpper);
                            ctx.lineTo(x, yLower);
                            ctx.stroke();

                            // Draw short horizontal caps at ends (light blue)
                            const capWidth = 8;
                            ctx.beginPath();
                            ctx.moveTo(x - capWidth, yUpper);
                            ctx.lineTo(x + capWidth, yUpper);
                            ctx.stroke();
                            ctx.beginPath();
                            ctx.moveTo(x - capWidth, yLower);
                            ctx.lineTo(x + capWidth, yLower);
                            ctx.stroke();
                        });
                    }
                };

                // Plugin to draw median squares offset to the right
                const medianPlugin = {
                    id: 'medianSquares',
                    afterDatasetsDraw(chart) {
                        const ctx = chart.ctx;
                        const xScale = chart.scales.x;
                        const yScale = chart.scales.y;

                        groupStats.forEach((group, idx) => {
                            const x = xScale.getPixelForValue(idx + 0.08); // Offset to right
                            const y = yScale.getPixelForValue(group.median);
                            const size = 5;

                            // Draw solid black square
                            ctx.fillStyle = '#000000';
                            ctx.fillRect(x - size, y - size, size * 2, size * 2);
                        });
                    }
                };
                
                // Calculate y-axis range based on CI bounds, not just means
                let minCIValue = Infinity;
                let maxCIValue = -Infinity;
                
                groupStats.forEach(g => {
                    minCIValue = Math.min(minCIValue, g.ciLower);
                    maxCIValue = Math.max(maxCIValue, g.ciUpper);
                });
                
                const ciRange = maxCIValue - minCIValue;
                const axisPadding = ciRange * 0.15; // 15% padding on each side
                const yMin = minCIValue - axisPadding;
                const yMax = maxCIValue + axisPadding;

                window.chartInstance = new Chart(ctx, {
                    type: 'scatter',
                    data: {
                        datasets: [
                            {
                                label: 'Mean (95% CI)',
                                data: groupStats.map((g, idx) => ({
                                    x: idx,
                                    y: g.mean
                                })),
                                pointRadius: 7,
                                pointBackgroundColor: '#FFFFFF',
                                pointBorderColor: '#000000',
                                pointBorderWidth: 2,
                                showLine: false,
                                tension: 0
                            },
                            {
                                label: 'Median',
                                data: [], // Empty - drawn by plugin
                                showLine: false,
                                pointRadius: 5,
                                pointStyle: 'rect',
                                pointBackgroundColor: '#000000',
                                pointBorderColor: '#000000'
                            }
                        ]
                    },
                    options: {
                        responsive: true,
                        maintainAspectRatio: false,
                        plugins: {
                            ciErrorBars: {},
                            medianSquares: {},
                            legend: {
                                display: true,
                                position: 'top',
                                labels: {
                                    font: { size: 12, family: 'Arial, sans-serif', weight: '500' },
                                    color: '#000000',
                                    usePointStyle: true,
                                    padding: 20,
                                    boxPadding: 8
                                },
                                align: 'center'
                            },
                            title: {
                                display: true,
                                text: 'Mean with 95% Confidence Interval and Median',
                                font: { size: 13, family: 'Arial, sans-serif', weight: 'bold' },
                                color: '#000000',
                                padding: { top: 10, bottom: 20 }
                            },
                            tooltip: {
                                backgroundColor: 'rgba(0, 0, 0, 0.85)',
                                padding: 12,
                                titleFont: { size: 12, family: 'Arial, sans-serif', weight: 'bold' },
                                bodyFont: { size: 11, family: 'Arial, sans-serif' },
                                borderColor: '#000000',
                                borderWidth: 1,
                                displayColors: false,
                                callbacks: {
                                    title: function(context) {
                                        return `Group ${groupStats[context[0].dataIndex].name}`;
                                    },
                                    label: function(context) {
                                        const idx = context.dataIndex;
                                        const group = groupStats[idx];
                                        if (context.dataset.label === 'Mean (95% CI)') {
                                            return `Mean: ${group.mean.toFixed(2)}\n95% CI: [${group.ciLower.toFixed(2)}, ${group.ciUpper.toFixed(2)}]\nMedian: ${group.median.toFixed(2)}`;
                                        }
                                        return '';
                                    },
                                    afterLabel: function(context) {
                                        const idx = context.dataIndex;
                                        const group = groupStats[idx];
                                        return `n = ${group.n}, SD = ${group.sd.toFixed(2)}`;
                                    }
                                }
                            }
                        },
                        scales: {
                            x: {
                                type: 'linear',
                                min: -0.5,
                                max: groupStats.length - 0.5,
                                ticks: {
                                    stepSize: 1,
                                    callback: function(value) {
                                        return groupStats[value] ? groupStats[value].name : '';
                                    },
                                    font: { size: 12, family: 'Arial, sans-serif', weight: '500' },
                                    color: '#000000'
                                },
                                title: {
                                    display: true,
                                    text: 'Group',
                                    font: { size: 12, family: 'Arial, sans-serif', weight: 'bold' },
                                    color: '#000000',
                                    padding: { top: 12 }
                                },
                                grid: {
                                    display: false,
                                    drawBorder: true,
                                    color: '#000000',
                                    lineWidth: 1
                                }
                            },
                            y: {
                                min: yMin,
                                max: yMax,
                                ticks: {
                                    font: { size: 11, family: 'Arial, sans-serif' },
                                    color: '#000000'
                                },
                                title: {
                                    display: true,
                                    text: 'Value',
                                    font: { size: 12, family: 'Arial, sans-serif', weight: 'bold' },
                                    color: '#000000',
                                    padding: { bottom: 12 }
                                },
                                grid: {
                                    display: true,
                                    drawBorder: true,
                                    color: '#E5E5E5',
                                    lineWidth: 1,
                                    drawTicks: true
                                }
                            }
                        }
                    },
                    plugins: [ciErrorBarsPlugin, medianPlugin]
                });
            }
        }

        function exportVisualization(elementId) {
            const element = document.getElementById(elementId);
            if (!element) return;

            html2canvas(element, {
                backgroundColor: '#ffffff',
                scale: 2,
                logging: false
            }).then(canvas => {
                const link = document.createElement('a');
                link.href = canvas.toDataURL('image/png');
                link.download = `${elementId}-${new Date().getTime()}.png`;
                link.click();
            });
        }

        function copyTableAsTSV(elementId) {
            const element = document.getElementById(elementId);
            if (!element) return;

            const table = element.querySelector('table');
            if (!table) return;

            let tsv = '';
            const rows = table.querySelectorAll('tr');

            rows.forEach(row => {
                const cells = row.querySelectorAll('th, td');
                const rowData = Array.from(cells).map(cell => cell.textContent.trim());
                tsv += rowData.join('\t') + '\n';
            });

            navigator.clipboard.writeText(tsv).then(() => {
                const button = event.target;
                const originalText = button.textContent;
                button.textContent = '✓ Copied!';
                setTimeout(() => {
                    button.textContent = originalText;
                }, 2000);
            }).catch(err => {
                console.error('Failed to copy:', err);
            });
        }

        // ============= Utility Functions =============
        function groupBy(data, key) {
            const result = {};
            data.forEach(item => {
                const keyValue = typeof key === 'function' ? key(item) : item[key];
                if (!result[keyValue]) result[keyValue] = [];
                result[keyValue].push(item);
            });
            return result;
        }
