// KCL
function updateKCL() {
  const i1=parseFloat(document.getElementById('kcl1').value), i2=parseFloat(document.getElementById('kcl2').value);
  const i3=parseFloat(document.getElementById('kcl3').value), i4=parseFloat(document.getElementById('kcl4').value);
  document.getElementById('kcl1v').textContent=i1.toFixed(1);
  document.getElementById('kcl2v').textContent=i2.toFixed(1);
  document.getElementById('kcl3v').textContent=i3.toFixed(1);
  document.getElementById('kcl4v').textContent=i4.toFixed(1);
  const sumIn=i1+i2, sumOut=i3+i4, balanced=Math.abs(sumIn-sumOut)<0.05;
  const sw=v=>Math.max(2,v*3);
  document.getElementById('kclSvg').innerHTML=`<svg width="220" height="160" viewBox="0 0 220 160">
<circle cx="110" cy="80" r="14" fill="#0891b2"/>
<line x1="20" y1="45" x2="96" y2="74" stroke="#10b981" stroke-width="${sw(i1)}" marker-end="url(#aG)" class="arrow-anim"/>
<line x1="20" y1="115" x2="96" y2="86" stroke="#10b981" stroke-width="${sw(i2)}" marker-end="url(#aG)" class="arrow-anim"/>
<line x1="124" y1="74" x2="200" y2="45" stroke="#ef4444" stroke-width="${sw(i3)}" marker-end="url(#aR)" class="arrow-anim"/>
<line x1="124" y1="86" x2="200" y2="115" stroke="#ef4444" stroke-width="${sw(i4)}" marker-end="url(#aR)" class="arrow-anim"/>
<text x="5" y="40" fill="#10b981" font-size="12">${i1.toFixed(1)}A</text>
<text x="5" y="125" fill="#10b981" font-size="12">${i2.toFixed(1)}A</text>
<text x="185" y="40" fill="#ef4444" font-size="12">${i3.toFixed(1)}A</text>
<text x="185" y="125" fill="#ef4444" font-size="12">${i4.toFixed(1)}A</text>
<defs><marker id="aG" markerWidth="8" markerHeight="8" refX="6" refY="4" orient="auto"><path d="M0,0 L8,4 L0,8 Z" fill="#10b981"/></marker><marker id="aR" markerWidth="8" markerHeight="8" refX="6" refY="4" orient="auto"><path d="M0,0 L8,4 L0,8 Z" fill="#ef4444"/></marker></defs></svg>`;
  const cls=balanced?'balanced':'not-balanced';
  const badge=balanced?'<span class="badge-ok">✓ Balanced Junction</span>':'<span class="badge-err">✗ Not Balanced</span>';
  document.getElementById('kclResult').className='result-box '+cls;
  document.getElementById('kclResult').innerHTML=`<p>Total In: <strong>${sumIn.toFixed(2)} A</strong> | Total Out: <strong>${sumOut.toFixed(2)} A</strong> | Difference: <strong>${Math.abs(sumIn-sumOut).toFixed(2)} A</strong></p><p style="margin-top:8px;">${badge}</p>${!balanced?`<p style="margin-top:6px;font-size:0.85rem;color:#94a3b8;">To balance: I₄ should be ${(sumIn-i3).toFixed(2)} A, or I₃ should be ${(sumIn-i4).toFixed(2)} A</p>`:''}`;
}

// KVL
function updateKVL() {
  const V=parseFloat(document.getElementById('kvlV').value);
  const R1=parseFloat(document.getElementById('kvlR1').value), R2=parseFloat(document.getElementById('kvlR2').value), R3=parseFloat(document.getElementById('kvlR3').value);
  document.getElementById('kvlVv').textContent=V;
  document.getElementById('kvlR1v').textContent=R1;
  document.getElementById('kvlR2v').textContent=R2;
  document.getElementById('kvlR3v').textContent=R3;

// Two-Loop
function updateTL() {
  const V1=parseFloat(document.getElementById('tlV1').value), V2=parseFloat(document.getElementById('tlV2').value);
  const R1=parseFloat(document.getElementById('tlR1').value), R2=parseFloat(document.getElementById('tlR2').value), R3=parseFloat(document.getElementById('tlR3').value);
  document.getElementById('tlV1v').textContent=V1; document.getElementById('tlV2v').textContent=V2;
  document.getElementById('tlR1v').textContent=R1; document.getElementById('tlR2v').textContent=R2; document.getElementById('tlR3v').textContent=R3;
  // Solve: V1 = I1(R1+R3) - I2*R3 ; V2 = I2(R2+R3) - I1*R3
  const a=R1+R3, b=-R3, c=-R3, d=R2+R3;
  const det=a*d-b*c;
  const I1=(V1*d-V2*b)/det, I2=(V2*a-V1*c)/det, I3=I1-I2;
  document.getElementById('tlResult').innerHTML=`
<p><strong>Loop Currents:</strong></p>
<div class="formula">I₁ = ${I1.toFixed(4)} A &nbsp;|&nbsp; I₂ = ${I2.toFixed(4)} A</div>
<div class="formula">Current through R₃: I₃ = I₁ - I₂ = ${I3.toFixed(4)} A ${I3<0?'(flows opposite to assumed direction)':''}</div>`;
  document.getElementById('tlSteps').innerHTML=`
<p><strong>Step 1: Given values</strong></p>
<div class="formula">V₁=${V1}V, V₂=${V2}V, R₁=${R1}Ω, R₂=${R2}Ω, R₃=${R3}Ω</div>
<p><strong>Step 2: KVL equations</strong></p>
<div class="formula">Left loop: V₁ - I₁R₁ - (I₁-I₂)R₃ = 0</div>
<div class="formula">Right loop: V₂ - I₂R₂ - (I₂-I₁)R₃ = 0</div>
<p><strong>Step 3: Simplify</strong></p>
<div class="formula">${V1} = I₁(${R1}+${R3}) - I₂(${R3}) → ${V1} = ${a}I₁ - ${R3}I₂</div>
<div class="formula">${V2} = I₂(${R2}+${R3}) - I₁(${R3}) → ${V2} = ${d}I₂ - ${R3}I₁</div>
<p><strong>Step 4: Solve (Cramer's rule)</strong></p>
<div class="formula">Determinant = (${a})(${d}) - (${b})(${c}) = ${det}</div>
<div class="formula">I₁ = (${V1}×${d} - ${V2}×${b}) / ${det} = ${I1.toFixed(4)} A</div>
<div class="formula">I₂ = (${V2}×${a} - ${V1}×${c}) / ${det} = ${I2.toFixed(4)} A</div>
<div class="formula">I₃ = I₁ - I₂ = ${I3.toFixed(4)} A</div>`;
}
updateTL();
